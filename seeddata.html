Object.entries(HAITI).forEach(([code, communes]) => {

});
return pool[Math.floor(Math.random() * pool.length)];
}

const log = document.getElementById('log');
const prog = document.getElementById('progress');

function addLog(msg, cls='info') {
const line = document.createElement('div');
line.className = cls;
line.textContent = new Date().toLocaleTimeString() + ' — ' + msg;
log.appendChild(line);
log.scrollTop = log.scrollHeight;
}

async function sendOne(data) {
data.is_real = 'false';
const p = new URLSearchParams();
Object.entries(data).forEach(([k,v]) => p.append(k, String(v)));
await fetch(URL, {method:'POST', body:p, mode:'no-cors'});
}

async function seed(type, count) {
const btns = document.querySelectorAll('.btn');
btns.forEach(b => b.disabled = true);
addLog(`Envoi de ${count} ${type}(s)...`, 'info');
for (let i = 0; i < count; i++) {
try {
const data = type === 'sentinelle' ? makeSentinelle(i) : makeCoach(i);
await sendOne(data);
addLog(`✓ ${type} ${i+1}/${count} — ${data.prenom} ${data.nom} (${data.commune||data.location_type})`, 'ok');
prog.textContent = `${i+1} / ${count} envoyé(s)`;
await new Promise(r => setTimeout(r, 600));
} catch(e) {
addLog(`✗ Erreur ${i+1}: ${e.message}`, 'err');
}
}
addLog(`✅ Terminé — ${count} ${type}(s) envoyé(s). Vérifiez le Google Sheet.`, 'ok');
prog.textContent = '';
btns.forEach(b => b.disabled = false);
// Auto-check after seeding
await new Promise(r => setTimeout(r, 2000));
checkSheet();
}

async function seedAll() {

async function checkSheet() {
addLog('Vérification du Sheet...', 'dim');
try {
const r = await fetch(URL + '?t=' + Date.now());
const d = await r.json();
addLog(`Sheet: ${d.sentinelles||0} sentinèl · ${d.coaches||0} coach`,
(d.sentinelles > 0 || d.coaches > 0) ? 'ok' : 'err');
if (d.sentinelles === 0 && d.coaches === 0) {
addLog('⚠ Sheet vide — les données ne parviennent pas au script.', 'err');
addLog('→ Vérifiez que le script v3 est bien déployé (Deploy → Manage → New version)', 'err');
}
} catch(e) {
addLog('Erreur de connexion au sheet: ' + e.message, 'err');
}
}

<script>
let lang='ht';
function L(l){
lang=l;
document.querySelectorAll('.lb').forEach(b=>{
const m={ht:'🇭🇹 KR',fr:'🇫🇷 FR',en:'🇺🇸 EN'};
b.classList.toggle('active',b.textContent.trim()===m[l]);
});
const t=STRINGS[l];
document.querySelectorAll('[data-s]').forEach(el=>{
const k=el.getAttribute('data-s');
if(t[k])el.innerHTML=t[k];
});
}

function fillCommunes(deptId,commId){
const dept=document.getElementById(deptId).value;
const sel=document.getElementById(commId);
sel.innerHTML='<option value="">—</option>';
if(dept&&HAITI[dept]){
HAITI[dept].sort().forEach(c=>{
const o=document.createElement('option');
o.value=COMMUNE_CODES[c]||c.substring(0,4).toUpperCase();
o.textContent=c;
sel.appendChild(o);
});
}
}

function populateDepts(){
const sel=document.getElementById('s-dept');
Object.keys(HAITI).sort().forEach(d=>{
const o=document.createElement('option');
o.value=DEPT_CODES[d]||d.substring(0,4).toUpperCase();
o.textContent=d;
sel.appendChild(o);
});
}

function switchTab(tab){
document.querySelectorAll('.ftab').forEach((b,i)=>b.classList.toggle('active',(i===0&&tab==='sentinelle')||(i===1&&tab==='coach')));
document.getElementById('panel-sentinelle').classList.toggle('active',tab==='sentinelle');
document.getElementById('panel-coach').classList.toggle('active',tab==='coach');
history.replaceState(null,'','#'+tab);
}

async function post(data){
try{
const p=new URLSearchParams();
Object.entries(data).forEach(([k,v])=>p.append(k,String(v)));
await fetch(SCRIPT_URL,{method:'POST',body:p,mode:'no-cors'});
}catch(e){}
}

async function submitForm(e,type){
e.preventDefault();
const fd=new FormData(e.target);
const data={type};
fd.forEach((v,k)=>{data[k]=v;});
// Add dept name alongside code
if(type==='sentinelle'){
const deptSel=document.getElementById('s-dept');
data.departement_nom=deptSel.options[deptSel.selectedIndex]?.text||'';
const commSel=document.getElementById('s-comm');
data.commune_nom=commSel.options[commSel.selectedIndex]?.text||'';
}
data.langue=lang;
data.source='website';
data.is_real='true';
await post(data);
e.target.style.display='none';
document.getElementById('suc-'+type[0]).style.display='block';
}

async function fetchStats(){
try{
const r=await fetch(SCRIPT_URL+'?t='+Date.now());
const d=await r.json();
document.getElementById('sb-s').textContent=d.sentinelles||0;
document.getElementById('sb-c').textContent=d.coaches||0;
}catch(e){}
}

// Handle anchor on load
window.addEventListener('load',()=>{
populateDepts();
L('ht');
fetchStats();
if(location.hash==='#coach') switchTab('coach');
});
</script>
